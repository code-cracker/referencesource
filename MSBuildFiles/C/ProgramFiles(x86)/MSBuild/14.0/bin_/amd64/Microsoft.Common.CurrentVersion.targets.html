    </ItemGroup>

      <Output TaskParameter="Filtered" ItemName="_CleanUniqueRemainingFileWrites"/>
    </RemoveDuplicates>
 
    <!-- Make sure the directory exists. -->
    <MakeDir Directories="$(IntermediateOutputPath)"/>
 
    <!-- Write new list of current files back to disk. -->

  </Target>
